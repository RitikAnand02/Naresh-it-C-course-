Write a C program to print square pattern program using * 



* * * * *

* * * * *

* * * * *

* * * * *

* * * * *
ans:
#include <stdio.h>
int main()
{
    int i, j, n;
    printf("Enter rows ");
    scanf("%d", &n);
    for (i = 1; i <= n; i++)
    {
        for (j = 1; j <= n; j++)
        {
            printf("* ");
        }
        printf("\n");
    }
}

Write a C program to print square pattern program 

  1 1 2 1 3 1

  1 2 2 2 3 2

  1 3 2 3 3 3 

  1 4 2 4 3 4

  1 5 2 5 3 5
ans:
#include <stdio.h>
int main()
{
    int i, j, n, m;
    printf("Enter row and columns ");
    scanf("%d%d", &n, &m);
    for (i = 1; i <= n; i++)
    {
        int k = 1;
        for (j = 1; j <= m; j++)
        {
            if (j % 2 == 0)
            {
                printf("%3d", i);
            }
            else
            {
                printf("%3d", k++);
            }
        }
        printf("\n");
    }
}

Write a C program to print square pattern program 

   

   5 10 15 20 25

   4 9  14 19 24

   3 8  13 18 23

   2 7  12 17 22

   1 6  11 16 21
ans:
#include <stdio.h>
int main()
{
    int i, j, n, k = 5;
    printf("Enter rows ");
    scanf("%d", &n);
    for (i = 5; i >= 1; i--)
    {
        k = i;
        for (j = 1; j <= n; j++)
        {
            printf("%3d", k);
            k += 5;
        }
        printf("\n");
    }
}
