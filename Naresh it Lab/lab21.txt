Write a program in C to display the cube of the number up to an integer.



Test Data :

Input number of terms : 5

Expected Output :

Number is : 1 and cube of the 1 is :1

Number is : 2 and cube of the 2 is :8

Number is : 3 and cube of the 3 is :27

Number is : 4 and cube of the 4 is :64

Number is : 5 and cube of the 5 is :125
ans:
#include <stdio.h>
int main()
{
    int n, i;
    printf("Enter the no:");
    scanf("%d", &n);
    for (i = 1; i <= n; i++)
    {
        printf("Number is : %d and cube of the %d is :%d\n", i, i, i * i * i);
    }
}

Write a C program to display the sum of n terms of even natural numbers.



Test Data :

Input number of terms : 5

Expected Output :

The even numbers are :2 4 6 8 10

The Sum of even Natural Number upto 5 terms : 30
ans:
#include <stdio.h>
int main()
{
    int n, i, s = 0;
    printf("Enter the no:");
    scanf("%d", &n);
    for (i = 1; i <= n * 2; i++)
    {
        if (i % 2 == 0)
        {
            s = s + i;
            printf("even no are=%d\n", i);
            i++;
        }
    }
    printf("sum of even natural no=%d", s);
}

Write a program in C to find the sum of the series 1 +11 + 111 + 1111 + .. n terms.



Test Data :

Input the number of terms : 5

Expected Output :

1 + 11 + 111 + 1111 + 11111

The Sum is : 12345
ans:
#include <stdio.h>
int main()
{
    int n, i, s = 0, sum = 0;
    printf("Enter the no:");
    scanf("%d", &n);
    for (i = 1; i <= n; i++)
    {
        s = (s * 10) + 1;
        printf("%d+", s);
        sum = sum + s;
    }
    printf("\b=%d", sum);
}

Write a program in C to display the n terms of a harmonic series and their sum.

1 -1/2 +1/3 - 1/4 + 1/5 -1/6... 1/n terms



Test Data :

Enter the value of n: 5

Harmonic Series: 1 - 1/2 + 1/3 - 1/4 + 1/5

Expected Output :

Sum of the series: 0.783333
ans:
#include <stdio.h>
int main()
{
    int n, i;
    float s = 1;
    printf("Enter a no ");
    scanf("%d", &n);
    printf("1-");
    for (i = 2; i <= n; i++)
    {
        if (i % 2 == 0)
            printf("1/%d+", i, s = s - 1.0 / i);
        else
            printf("1/%d-", i, s = s + 1.0 / i);
    }

    printf("\b=%.6f", s);
}
